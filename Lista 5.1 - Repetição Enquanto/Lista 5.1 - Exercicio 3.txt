// o valor do desconto e a gratificação do funcionário

// Os nomes devem ser lidos
// até que a sequência XXXX seja digitada.

var
salario, desconto, gratificacao, salLiquido : real
nome : caracter

inicio
escreva ("Digite o nome: ")
leia (nome)

salLiquido <- 0
enquanto (nome<>"xxxx") faca
escreva ("Digite o salario bruto: R$")
leia (salario)
escreva ("Digite o valor de desconto: R$")
leia (desconto)
escreva ("Digite a gratificação: R$")
leia (gratificacao)
salLiquido <- salario - desconto + gratificacao
escreval ("O salário líquido é de: R$",salLiquido)

leia (nome)
