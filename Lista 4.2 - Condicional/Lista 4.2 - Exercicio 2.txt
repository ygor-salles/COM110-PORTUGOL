algoritmo "Lista 4.2 - Exercicio 2"

// Considere que a empresa mencionada no exercício anterior tenha decidido mudar a regra da seguinte maneira.
//  a. Se quantidade mensal de peças produzida pelo funcionário > 120 durante os últimos 12 meses, o funcionário ganha 5% do valor líquido do salário.
//  b. Senão, se o funcionário não faltou durante os 12 últimos meses, ganha 2% do valor líquido do salário.
//  Elabore um algoritmo que receba a quantidade de falta de um funcionário e a quantidade de peças produzidas nos últimos 12 meses e retorne o valor
//  do salário mais o abono considerando a nova regra.

var
falta, peca : inteiro
sal, abonoPeca, abonoPresenca, salTotal : real

inicio
escreva ("Digite a quantidade de peças produzidas no ultimos 12 meses: ")
leia (peca)
escreva ("Digite a quantidade de faltas do funcionario no ultimos 12 meses: ")
leia (falta)
escreva ("Digite o valor do salário do funcionário: ")
leia (sal)

abonoPeca <- (sal*0.05)
abonoPresenca <- (sal*0.02)

se ((peca > 120) e (falta = 0)) entao
salTotal <- sal + abonoPeca + abonoPresenca
escreval ("O valor do salário mais o abono total é de: R$",salTotal)
senao
     se (peca > 120) entao
     salTotal <- sal + abonoPeca
     escreval ("O valor do salário mais o abono é de: R$",salTotal)
     senao
          se (falta = 0) entao
          salTotal <- sal + abonoPresenca
          escreval ("O valor do salário mais o abono é de: R$",salTotal)
          senao
          salTotal <- sal
          escreval ("O valor do salário total é de :R$",salTotal)
          fimse
     fimse
fimse
