// Escreva um algoritmo para calcular N! (fatorial de N), sendo que o valor inteiro de N é
// fornecido pelo usuário. Sabe-se que: N! = 1 * 2 * 3 * ... * (N-1) * N; 0! = 1, por
// definição

var
i, n, fatorial : inteiro

inicio
escreva ("Digite um valor inteiro: ")  //Pedir para o usuário digitar um valor antes do calculo
leia (n)
fatorial <- 1  //A variável deve inicializar com 1 para multiplicar com o valor que o usuário digitou e ser o mesmo resultado
para i de n ate 1 passo -1 faca //A variável i assumirá o valor de n, a cada loop. Ex: 5; 4; 3...
     fatorial <- fatorial*i  //No primeiro loop o "fatorial"=1 * contador (usuário digitou)-1 no 2º loop, e assim por diante ate chegar no 1
fimpara
escreval ("O valor de ",n," fatorial é: ",fatorial)
